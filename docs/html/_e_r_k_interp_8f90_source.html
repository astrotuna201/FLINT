28  module subroutine erk_interp(me, xarr, yarr, saveinterpcoeffs)
29 
30  implicit none
31 
32  class(ERK_class), intent(inout) :: me
33 
34  real(WP), dimension(1:), intent(in) :: Xarr
35  real(WP), dimension(size(me%InterpStates),size(Xarr)), intent(out) :: Yarr
36  logical, intent(in), optional :: SaveInterpCoeffs
37 
38  logical :: DeallocMem
39  integer :: n, ctr, X0loc, X0start
40  real(WP) :: h, hSign, X0
41 
42  ! if interpolation mode is not on then return with an error
43  if (.NOT. me%InterpOn) then
44  me%status = flint_error_intp_off
45  return
46  end if
47 
48  ! if user does not explicitly say to keep the internal memory for future interpolations
49  ! then free this memory.
50  deallocmem = .true.
51  if (present(saveinterpcoeffs)) deallocmem = (.NOT. saveinterpcoeffs)
52 
53  ! If empty array then just return without error
54  n = size(xarr)
55  if (n < 1) return
56 
57  ! Xarr must be monotonically increasing or decreasing with no repetitions
58  hsign = sign(1.0_wp, (me%Xint(me%AcceptedSteps+1)-me%Xint(1)))
59  block
60  real(WP), dimension(size(Xarr)-1) :: tXarr
61  txarr = xarr(1:(n-1)) - xarr(2:n)
62  if (any((hsign*txarr) >= 0)) then
63  me%status = flint_error_intp_array
64  return
65  end if
66  end block
67 
68  ! Xarr elements must lie between the initial and final values of X
69  if (hsign*(xarr(1)-me%Xint(1)) < 0 .OR. hsign*(xarr(n)-me%Xint(me%AcceptedSteps+1))>0) then
70  me%status = flint_error_intp_array
71  return
72  end if
73 
74  ! Binary search seems to be slower than the linear search. May be
75  ! because Xarr is also a sorted array
76  ! Xarrloc = BinarySearch(Xarr, me%Xint)
77 
78  ! Main Interpolation loop
79  x0start = 1
80 
81  do concurrent(ctr = 1:n)
82 
83  ! Find X0loc s.t. Xarr(ctr) belongs to ( Xint(X0loc-1), Xint(X0loc) ]
84  do while (x0start <= me%AcceptedSteps)
85  x0start = x0start + 1
86  if (hsign*(me%Xint(x0start)-xarr(ctr)) >= 0) then
87  x0loc = x0start - 1
88  exit
89  end if
90  end do
91 
92  if (x0loc <= me%AcceptedSteps) then
93  x0 = me%Xint(x0loc)
94  h = me%Xint(x0loc+1)-me%Xint(x0loc)
95  else
96  ! There is a problem, we should never reach here!
97  x0 = xarr(ctr)
98  h = 0.0
99  end if
100 
101  ! compute the interpolation polynomial
102  yarr(:, ctr) = interpy(size(me%InterpStates), xarr(ctr), x0, h, &
103  me%pstar, me%Bip(:,:,x0loc))
104 
105  ! prepare for the next point
106  x0start = x0loc
107  end do
108 
109 
110  ! job done, deallocate memory
111  if (deallocmem .EQV. .true.) then
112  deallocate(me%Xint)
113  !deallocate(me%Yint)
114  deallocate(me%Bip)
115  end if
116 
117 
118  end subroutine erk_interp

121  pure module function interpy(np, x, x0, h, pstar, bcoeffs) result(yip)
122  integer, intent(in) :: np
123  real(WP), intent(in) :: X, X0, h
124  integer, intent(in) :: pstar
125  real(WP), dimension(np,0:pstar), intent(in) :: bCoeffs
126  real(WP), dimension(np) :: Yip
127 
128  real(WP) :: theta
129  integer :: k
130 
131  ! compute the interpolation polynomial
132  theta = (x - x0)/h
133  yip = 0.0_wp
134  do k = 0,pstar
135  yip = yip + bcoeffs(:,k)*(theta**k)
136  end do
137  end function interpy
