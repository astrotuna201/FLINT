57  ! reset the status
58  me%status = flint_success
59 
60  ! assign the method
61  if (present(method)) me%Method = method
62 
63  ! Pointer to the user-supplied Diff Eq system object
64  me%pDiffEqSys => de
65 
66  ! user must specify maximum number of steps allowed to take,
67  ! we will allocate memory based on this number
68  if (maxsteps <= 0) then
69  me%status = flint_error_params
70  return
71  else
72  me%MaxSteps = maxsteps
73  end if
74 
75  ! Check if tolerances are scalars or vectors?
76  if (size(atol) == 1 .AND. size(rtol) == 1) then
77  me%IsScalarTol = .true.
78  me%ATol = atol
79  me%RTol = rtol
80  else if (size(atol) == me%pDiffEqSys%n .AND. size(rtol) == me%pDiffEqSys%n) then
81  me%IsScalarTol = .false.
82  me%ATol = atol
83  me%RTol = rtol
84  else
85  me%status = flint_error_params
86  return
87  end if
88 
89  ! Step-size related options
90 
91  ! Minimum-step size: if 0 then it is equal to eps
92  if (present(minstepsize)) then
93  me%MinStepSize = abs(minstepsize)
94  else
95  me%MinStepSize = eps
96  end if
97 
98  ! Maximum-step size: A "0" value means that it will be set
99  ! equal to (Xf - X0)
100  if (present(maxstepsize)) then
101  me%MaxStepSize = abs(maxstepsize)
102  else
103  me%MaxStepSize = 0.0_wp
104  end if
105 
106  ! Method-specific step-size computation tuning parameters
107  ! Ask yourself before specifying: do you know what you are doing?
108  if (present(stepszparams)) then
109  me%StepSzParams = stepszparams
110  else
111  select case (me%method)
112  case (erk_dop853,erk_dop54,erk_verner98r, erk_verner65e)
113  me%StepSzParams = [sf, sfmin, sfmax, beta, hbyhoptold]
114  case default
115  ! Not implemented yet
116  end select
117 
118  end if
119 
120  ! Deallocate first before any allocation of interpolation space
121  call intpmemdeallocate()
122  if (present(interpon)) then
123  if (interpon) then
124  me%InterpOn = .true.
125  else
126  me%InterpOn = .false.
127  end if
128  end if
129 
130  ! Event related parameters start from here
131 
132  ! By default, event detection will be disabled
133  if (present(eventson)) me%EventsOn = eventson
134 
135  if (me%EventsOn) then
136  ! For events, we need to interpolate all the states
137  me%InterpStates = [integer :: (doI,doI=1,me%pDiffEqSys%n)]
138 
139  ! user-specified event step size
140  if (present(eventstepsz)) then
141  if (size(eventstepsz) /= me%pDiffEqSys%m) then
142  me%status = flint_error_dimension
143  return
144  else
145  ! make sure event step sz is always positive
146  me%EventStepSz = abs(eventstepsz)
147  end if
148  else
149  ! By default, events will be checked at the integrator's steps
150  me%EventStepSz = [integer :: (0,doI=1,me%pDiffEqSys%m)]
151  end if
152 
153  ! user-specified event options
154  if (present(eventoptions)) then
155  if (size(eventoptions) /= me%pDiffEqSys%m) then
156  me%status = flint_error_dimension
157  return
158  else if (any(eventoptions < flint_eventoption_rootfinding) &
159  .OR. any(eventoptions > flint_eventoption_stepend)) then
160  me%status = flint_error_eventparams
161  return
162  else

164  end if
165  else
166  ! By default, events will be located using root-finding
167  me%EventOptions = &
168  [integer :: (FLINT_EVENTOPTION_ROOTFINDING,doI=1,me%pDiffEqSys%m)]
169  end if
170 
171  ! event tolerance option
172  if (present(eventtol)) then
173  if (size(eventtol) == me%pDiffEqSys%m) then
174  me%ETol = abs(eventtol)
175  else
176  me%status = flint_error_dimension
177  return
178  end if
179  else
180  me%ETol = [real(wp) :: (default_eventtol,doi=1,me%pDiffEqSys%m)]
181  end if
182 
183  end if
184 
185 
186 
187 
188 
189  ! choose method-specific parameters
190  select case (me%method)
191 
192  case (erk_dop853)
193  me%s = dop853_s
194  me%sint = dop853_sint
195  me%a(1:int((me%s-1)/2.0*me%s)) = dop853_a
196  me%b(1:me%sint) = dop853_b
197  me%c(2:me%s) = dop853_c
198  me%e(1:me%sint) = dop853_e
199  me%IsFSALMethod = dop853_fsal
200 
201  if (me%InterpOn .OR. me%EventsOn) then
202  me%pstar = dop853_pstar
203  me%dinz(1:size(dop853_di_nz)+1) = [dop853_di_nz, -1]
204  me%d(1:size(dop853_di_nz),1:size(dop853_dj_nz)) = dop853_d
205  end if
206 
207  case (erk_verner98r)
208  me%s = verner98r_s
209  me%sint = verner98r_sint
210  me%a(1:int((me%s-1)/2.0*me%s)) = verner98r_a
211  me%b(1:me%sint) = verner98r_b
212  me%c(2:me%s) = verner98r_c
213  me%e(1:me%sint) = verner98r_e
214  me%IsFSALMethod = verner98r_fsal
215 
216  if (me%InterpOn .OR. me%EventsOn) then
217  me%pstar = verner98r_pstar
218  me%dinz(1:size(verner98r_di_nz)+1) = [verner98r_di_nz, -1]
219  me%d(1:size(verner98r_di_nz),1:verner98r_pstar) = verner98r_d
220  end if
221 
222  case (erk_verner65e)
223  me%s = verner65e_s
224  me%sint = verner65e_sint
225  me%a(1:int((me%s-1)/2.0*me%s)) = verner65e_a
226  me%b(1:me%sint) = verner65e_b
227  me%c(2:me%s) = verner65e_c
228  me%e(1:me%sint) = verner65e_e
229  me%IsFSALMethod = verner65e_fsal
230 
231  if (me%InterpOn .OR. me%EventsOn) then
232  me%pstar = verner65e_pstar
233  me%dinz(1:size(verner65e_di_nz)+1) = [verner65e_di_nz, -1]
234  me%d(1:size(verner65e_di_nz),1:verner65e_pstar) = verner65e_d
235  end if
236 
237  case (erk_dop54)
238  me%s = dop54_s
239  me%sint = dop54_sint
240  me%a(1:int((me%s-1)/2.0*me%s)) = dop54_a
241  me%b(1:me%sint) = dop54_b
242  me%c(2:me%s) = dop54_c
243  me%e(1:me%sint) = dop54_e
244  me%IsFSALMethod = dop54_fsal
245 
246  if (me%InterpOn .OR. me%EventsOn) then
247  me%pstar = dop54_pstar
248  me%dinz(1:size(dop54_di_nz)+1) = [dop54_di_nz, -1]
249  me%d(1:size(dop54_di_nz),1:2) = dop54_d
250  end if
251 
252  case default
253  ! We will never be friends, so why are you here?
254  end select
255 
260 
261  if (me%InterpOn) then
262  ! check if dense output states are specified, else use all
263  ! If events are enabled, we save the coefficients for dense output
264  ! for all states by default
265  if (present(interpstates) .AND. (.Not. eventson)) then
266  ! check for valid components
267  if (size(interpstates) < 1 .OR. size(interpstates) > me%pDiffEqSys%n &
268  .OR. any(interpstates < 1) .OR. any(interpstates > me%pDiffEqSys%n)) then
269  me%status = flint_error_intpstates
270  else
271  me%InterpStates = interpstates
272  me%status = flint_success
273  end if
274  else
275  ! use all states for dense output
276  me%InterpStates = [integer :: (doI,doI=1,me%pDiffEqSys%n)]
277  me%status = flint_success
278  end if
279 
280  if (me%status /= flint_success) return
281 
282  ! Now allocate storage for internal steps as well as interpolation coefficients
283  call intpmemallocate()
284 
285  if (me%status /= flint_success) return
286  end if

296  !if (status /= 0) me%status = FLINT_ERROR_MEMALLOC
297 
298  allocate(me%Bip(size(me%InterpStates), 0:me%pstar, me%MaxSteps+1), stat=status)
299  if (status /= 0) me%status = flint_error_memalloc
300 
301  end subroutine
302

304 
305  if (allocated(me%Xint)) deallocate(me%Xint, stat = status)
