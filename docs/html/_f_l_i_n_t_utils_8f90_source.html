57  subroutine root(a, b, ya, yb, tol, f, x, fx, niter, error)
58 
59  implicit none
60 
61  real(WP), intent(in) :: a
62  real(WP), intent(in) :: b
63  real(WP), intent(in) :: ya
64  real(WP), intent(in) :: yb
65  real(WP), intent(in) :: tol
66  procedure(func) :: f
67  real(WP), intent(out) :: x
68  real(WP), intent(out) :: fx
69  integer, intent(out) :: niter
70  integer, intent(out) :: error
71 
72  real(WP) :: aa, bb, cc, fa, fb, fc, xm, ss, dd, ee, pp, qq, rr
73 
74  real(WP) :: tolx
75 
76  real(WP), parameter :: NEARZERO = 1.0e-20_wp
77 
78  logical :: done
79  integer :: itr
80 
81  aa = a
82  bb = b
83  fa = ya
84  fb = yb
85 
86  error = 0
87  niter = 0
88 
89  !! test
90  !fa = f(a)
91  !fb = f(b)
92 
93  ! Trivial cases
94  if (abs(fa) <= nearzero) then
95  x = aa
96  fx = fa
97  return
98  else if (abs(fb) <= nearzero) then
99  x = bb
100  fx = fb
101  return
102  end if
103 
104  ! if root is not bracketed then return error
105  if ((fa>0 .AND. fb>0) .OR. (fa<0 .AND. fb<0)) then
106  error = 1
107  niter = 0
108  return
109  end if
110 
111  ! algorithm start here
112  fc = fb
113  done = .false.
114  itr = 0
115  do while (done .EQV. .false. .AND. itr < max_iterations)
116 
117  ! if root is NOT bracketed between C and B, then C = A
118  if ((fc>0 .AND. fb>0) .OR. (fc<0 .AND. fb<0)) then
119  cc = aa
120  fc = fa
121  dd = bb - aa
122  ee = dd
123  end if
124  ! Make sure |f(c)| is bigger than |f(b)|
125  if (abs(fc) < abs(fb)) then
126  aa = bb
127  bb = cc
128  cc = aa
129  fa = fb
130  fb = fc
131  fc = fa
132  end if
133  ! tolerance on the independent variable
134  tolx = 2.0*eps*abs(bb) + 0.5*tol
135  ! middle point
136  xm = 0.5*(cc - bb)
137 
138  if (abs(xm) <= tolx .OR. abs(fa) < nearzero) then
139  ! root has been found
140  x = bb
141  done = .true.
142  fx = f(x)
143  else
144  if (abs(ee) >= tolx .AND. abs(fa) > abs(fb)) then
145  ss = fb/fa ! make sure |ss| < 1
146  if (abs(aa - cc) < nearzero) then
147  ! linear interpolation
148  pp = 2.0*xm*ss
149  qq = 1.0 - ss
150  else
151  ! Inverse quadratic interpolation
152  qq = fa/fc
153  rr = fb/fc
154  pp = ss*(2.0*xm*qq*(qq-rr)-(bb-aa)*(rr-1.0))
155  qq = (qq - 1.0)*(rr - 1.0)*(ss - 1.0)
156  end if
157 
158  if (pp > nearzero) qq = -qq
159  pp = abs(pp)
160  if ((2.0*pp) < min(3.0*xm*qq-abs(tolx*qq), abs(ee*qq))) then
161  ! use linear or inverse quadratic interpolation
162  ee = dd
163  dd = pp/qq
164  else
165  ! use bisection
166  dd = xm
167  ee = dd
168  end if
169  else
170  ! use bisection
171  dd = xm
172  ee = dd
173  end if
174 
175  aa = bb
176  fa = fb
177  if (abs(dd) > tolx) then
178  bb = bb + dd
179  else
180  ! Correction term too small, advance by at least tolx
181  if (xm > 0) then
182  bb = bb + abs(tolx)
183  else
184  bb = bb - abs(tolx)
185  end if
186  end if
187  fb = f(bb)
188  itr = itr + 1
189  end if
190  end do
191 
192  if (itr >= max_iterations) error = 2
193  niter = itr
194 
195  end subroutine root

202  pure function binarysearch(SortedX, SortedDataArray)
203 
204  !import :: WP
205 
206  implicit none
207 
208  real(wp), dimension(1:), intent(in) :: sortedx
209  real(wp), dimension(:), intent(in) :: sorteddataarray
210  integer, dimension(size(SortedX)) :: binarysearch
211 
212  integer :: lb, ub, a, b, m, n, ctr, xloc
213  real(wp) :: x
214 
215  ! initialize
216  lb = 1
217  ub = size(sorteddataarray)
218  m = lb
219 
220  ! for each element
221  do ctr = 1, size(sortedx)
222 
223  ! element to search for
224  x = sortedx(ctr)
225 
226  ! current interval. Note we assume that the array of the elements
227  ! to search for is also a sorted array.
228  a = m
229  b = ub
230 
231  ! loop for finding the element
232  do
233  ! size of the current interval
234  n = b - a + 1
235 
236  ! mid-point
237  m = floor((n+1)/2.0) + (a-1)
238 
239  if (x < sorteddataarray(m)) then
240  ! search the lower half
241  b = m
242  cycle
243  elseif (x > sorteddataarray(m) .AND. (m+1) /= b) then
244  ! search the upper half
245  a = m
246  cycle
247  else
248  ! We either have found the exact match
249  ! Or x lies in between Data(m) and Data(m+1)
250  xloc = m
251  exit
252  end if
253  end do
254 
255  ! save the location of the current element
256  binarysearch(ctr) = xloc
257  end do
258 
259  end function binarysearch
