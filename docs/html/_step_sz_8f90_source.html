78 
79  ! Algorithm starts here
80 
81  d0 = norm2(y0/sc)
82  d1 = norm2(f0/sc)
83 
84  ! first guess
85  if (d0 <= 1.0e-5 .or. d1 <= 1.0e-5) then
86  h0 = 1.0e-6_wp
87  else
88  h0 = 0.01_wp*d0/d1
89  end if
90 
91  h0 = hsign*min(h0, hmax)
92 
93  ! Explicit Euler step
94  f1 = pdiffeqsys%F((x0 + h0), (y0 + h0*f0), params)
95 
96  ! 2nd derivative approximation
97  d2 = norm2((f1 - f0)/sc)/h0
98 
99  ! second guess based on 2nd drivative
100  dmax = max(d1, abs(d2))
101 
106  if (dmax <= 1.0e-15 ) then
107  h1 = max(1.0e-6_wp, abs(h0)*1.0e-3_wp)
108  else
109  h1 = (0.01_wp/dmax)**(1.0/(p))
110  end if
111 
112  ! choose the final value for the initial step size
113  h = hsign*min(100.0_wp*abs(h0), h1, hmax)
114 
115  ! Total number of calls we made to F
116  fcalls = 1
117 
118  end subroutine stepsz0hairer
